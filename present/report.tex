\begin{frame}

   \frametitle{Уравнение осцилляций в среде}

   Уравнение нейтринных осцилляций в среде имеет вид
  \begin{equation*}
    \imath\frac{\dd\psi}{\dd\xi}=[H_{0}+\nu(\xi)W]\psi,
    \qquad
    \onslide<2->
    W=
    \begin{pmatrix}
      c_{13}^{2}c_{12}^{2} & c_{12}s_{12}c_{13}^{2} & c_{12}c_{13}s_{13}\\
      c_{12}s_{12}c_{13}^{2} & s_{12}^{2}c_{13}^{2} & s_{12}c_{13}s_{13}\\
      c_{12}s_{13}c_{13} & s_{12}c_{13}s_{13} & s_{13}^{2}
    \end{pmatrix}.
  \end{equation*}

   \onslide<3->
  Функция профиля плотности с современными значениями параметров
  \begin{equation*}
    v(r)=\gamma\exp^{-\eta\xi(r)}, \qquad \xi(r)=\frac{r}{R_{0}},
    \quad \gamma=6,5956\cdot10^{4}, \quad \eta=10,54,
  \end{equation*}

  \onslide<4->
 Современные значения углов смешивания
 \begin{align*}
   \sin^{2}{\theta_{12}}=0,307_{-0,012}^{+0,013}\\
   \sin^{2}{\theta_{23}}=0,417_{-0,028}^{+0,025}\\
   \sin^{2}{\theta_{13}}=0,0212\pm 0,0008
 \end{align*}


\end{frame}
